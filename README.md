### format

```json
{
  // a list of posts
  "posts": [
    {
      // date the post was comitted / posted
      "dateAdded": "",
      // slug generated from the date
      "slug": "",
      // list of content objects in the posts
      "content": [
        {
          // title of the content
          "title": "new framework",
          // url
          "url": "",
          // category of the content
          "category": "TS",
          // description based on the content in the url
          "description": ""
        }
      ]
    }
  ],
  // list of content categories
  "categories": ["TS", "PY", "JS"]
}
```
